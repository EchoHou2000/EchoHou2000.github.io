                                            public class TreeNode{
    int val;
    TreeNode left;
    TreeNode right;
    TreeNode(){}
    TreeNode(int val){
        this.val = val;
    }
    TreeNode(int val,TreeNode left,TreeNode right){
        this.val = val;
        this.left = left;
        this.right = right;
    }
}

public class Solution{
    // è¿™ä¸ªå‡½æ•°éšå«ç€å›æº¯çš„ç»†èŠ‚ï¼Œå› ä¸ºæ˜¯æŠŠtargetSumä½œä¸ºå‚æ•°æ•´ä½“ä¼ è¿›å»çš„
    // å‡½æ•°ç»“æŸåï¼ŒtargetSumçš„å€¼å¹¶æ²¡æœ‰æ”¹å˜
    public boolean hasPathSum(TreeNode root,int targetSum){
        if(root == null){
            return false;
        }
        targetSum -= root.val;
        if(root.left == null && root.right == null){
            return targetSum == 0;
        }

        if(root.left != null){
            if(hasPathSum(root.left,targetSum)){
                return true;
            }
        }
        if(root.right != null){
            if(hasPathSum(root.right,targetSum)){
                return true;
            }
        }
        return false;
    }
}
